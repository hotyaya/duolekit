CREATE SEQUENCE GEN_TAGID;
ALTER SEQUENCE GEN_TAGID RESTART WITH 0;

SET TERM ^ ;

CREATE TRIGGER TRIG_TAG_ID FOR TAG
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN

IF (NEW.tagid IS NULL) THEN
  NEW.tagid = GEN_ID(GEN_TAGID,1);
  
END^

SET TERM ; ^


--唯一索引
ALTER TABLE TAG
  ADD CONSTRAINT TAG_INDEX01
  UNIQUE (NAME);
